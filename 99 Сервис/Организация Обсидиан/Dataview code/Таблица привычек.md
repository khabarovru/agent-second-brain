```dataviewjs
// === Настройки ===
const FOLDER = "10 Организация Обсидиан/Привычки";

// === Получаем все привычки из папки ===
const habitPages = dv.pages(`"${FOLDER}"`);

// === Текущий год ===
const now = new Date();
const currentYear = now.getFullYear();
const startOfYear = new Date(currentYear, 0, 1);
const today = new Date();
const totalDaysInYear = Math.floor((today - startOfYear) / (1000 * 60 * 60 * 24)) + 1;

// === Функция для расчета streak ===
function calculateStreak(entries, year) {
    const yearEntries = entries.filter(dateStr => {
        const date = new Date(dateStr);
        return date.getFullYear() === year;
    });
    
    const entryDates = new Set(yearEntries.map(dateStr => {
        const date = new Date(dateStr);
        return date.toISOString().split('T')[0];
    }));
    
    let currentStreak = 0;
    let recordStreak = 0; // Для отслеживания рекордного значения
    const currentDate = new Date(startOfYear);
    const streakData = [];
    
    for (let i = 0; i < totalDaysInYear; i++) {
        const dateStr = currentDate.toISOString().split('T')[0];
        
        if (entryDates.has(dateStr)) {
            currentStreak = currentStreak + 1;
            // Обновляем рекорд, если текущий streak превышает его
            if (currentStreak > recordStreak) {
                recordStreak = currentStreak;
            }
        } else {
            currentStreak = Math.max(currentStreak - 1, 0);
        }
        
        streakData.push({
            date: dateStr,
            value: currentStreak,
            completed: entryDates.has(dateStr)
        });
        
        currentDate.setDate(currentDate.getDate() + 1);
    }
    
    return {
        currentStreak: streakData[streakData.length - 1].value,
        recordStreak: recordStreak,
        completedDays: yearEntries.length,
        missedDays: totalDaysInYear - yearEntries.length,
        streakData: streakData
    };
}

// === Обрабатываем каждую привычку (исключая "Привычки") ===
const habitData = [];
for (const page of habitPages) {
    // Исключаем привычку с названием "Привычки"
    if (page.file.name === "Привычки") continue;
    
    const entries = page.entries || [];
    const streakInfo = calculateStreak(entries, currentYear);
    
    habitData.push({
        name: page.file.name,
        currentStreak: streakInfo.currentStreak,
        recordStreak: streakInfo.recordStreak,
        completedDays: streakInfo.completedDays,
        missedDays: streakInfo.missedDays,
        streakData: streakInfo.streakData,
        completionRate: Math.round((streakInfo.completedDays / totalDaysInYear) * 100)
    });
}

// === Создаем агрегированную привычку "Привычки" ===
if (habitData.length > 0) {
    // Суммируем значения всех привычек
    const totalCompletedDays = habitData.reduce((sum, h) => sum + h.completedDays, 0);
    const totalMissedDays = habitData.reduce((sum, h) => sum + h.missedDays, 0);
    const totalCurrentStreak = habitData.reduce((sum, h) => sum + h.currentStreak, 0);
    const totalRecordStreak = habitData.reduce((sum, h) => sum + h.recordStreak, 0);
    
    // Рассчитываем общий процент выполнения
    const totalDays = totalCompletedDays + totalMissedDays;
    const totalCompletionRate = totalDays > 0 ? Math.round((totalCompletedDays / totalDays) * 100) : 0;
    
    // Добавляем агрегированную привычку
    habitData.push({
        name: "Привычки",
        currentStreak: totalCurrentStreak,
        recordStreak: totalRecordStreak,
        completedDays: totalCompletedDays,
        missedDays: totalMissedDays,
        completionRate: totalCompletionRate,
        isAggregated: true
    });
}

// === Создаем табличное представление ===
if (habitData.length === 0) {
    dv.paragraph("❌ Нет данных о привычках.");
} else {
    // Сортируем по текущему streak (по убыванию), агрегированная привычка будет в конце
    habitData.sort((a, b) => {
        if (a.name === "Привычки") return 1;
        if (b.name === "Привычки") return -1;
        return b.currentStreak - a.currentStreak;
    });
    
    // === Визуализация в виде таблицы ===
    dv.table(
        ["Привычка", "Выполнено дней", "Пропущено дней", "Текущий Streak", "Процент выполнения"],
        habitData.map(h => {
            const isAggregated = h.name === "Привычки";
            // Форматируем streak как "текущий/рекорд"
            const streakDisplay = isAggregated 
                ? `**${h.currentStreak}/${h.recordStreak}**` 
                : `${h.currentStreak}/${h.recordStreak}`;
                
            return [
                isAggregated ? `**${h.name}**` : h.name,
                isAggregated ? `**${h.completedDays}**` : h.completedDays.toString(),
                isAggregated ? `**${h.missedDays}**` : h.missedDays.toString(),
                streakDisplay,
                isAggregated ? `**${h.completionRate}%**` : `${h.completionRate}%`
            ];
        })
    );
}
```
