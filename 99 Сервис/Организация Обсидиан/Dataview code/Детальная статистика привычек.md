> // === Настройки ===
> const FOLDER = "10 Организация Обсидиан/Привычки";
> const TARGET_YEAR = 2025; // Укажите нужный год здесь
> 
> // === Получаем все привычки из папки ===
> const habitPages = dv.pages(`"${FOLDER}"`);
> 
> // === Функция для форматирования даты ===
> function formatDate(dateString) {
>     const date = new Date(dateString);
>     const options = { day: 'numeric', month: 'long', year: 'numeric' };
>     return date.toLocaleDateString('ru-RU', options);
> }
> 
> // === Функция для преобразования дней недели в числа ===
> function weekdayToNumber(weekday) {
>     const daysMap = {
>         "понедельник": 1, "вторник": 2, "среда": 3, "четверг": 4,
>         "пятница": 5, "суббота": 6, "воскресенье": 0
>     };
>     return daysMap[weekday.toLowerCase()] ?? null;
> }
> 
> // === Функция для получения всех дней после даты начала ===
> function getAllDaysAfterStart(year, startDate) {
>     const start = startDate ? new Date(startDate) : new Date(year, 0, 1);
>     const end = new Date(year, 11, 31);
>     const today = new Date();
>     const endDate = year === today.getFullYear() ? today : end;
>     
>     let days = [];
>     const current = new Date(start);
>     
>     while (current <= endDate) {
>         days.push(new Date(current));
>         current.setDate(current.getDate() + 1);
>     }
>     
>     return days;
> }
> 
> // === Функция для получения релевантных дней (для подсчета %) ===
> function getRelevantDays(year, startDate, weekdays) {
>     const start = startDate ? new Date(startDate) : new Date(year, 0, 1);
>     const end = new Date(year, 11, 31);
>     const today = new Date();
>     const endDate = year === today.getFullYear() ? today : end;
>     
>     let relevantDays = [];
>     const current = new Date(start);
>     
>     const targetWeekdays = weekdays && weekdays.length > 0 
>         ? weekdays.map(w => weekdayToNumber(w)).filter(w => w !== null)
>         : [0, 1, 2, 3, 4, 5, 6];
>     
>     while (current <= endDate) {
>         if (targetWeekdays.includes(current.getDay())) {
>             relevantDays.push(new Date(current));
>         }
>         current.setDate(current.getDate() + 1);
>     }
>     
>     return relevantDays;
> }
> 
> // === Функция для расчета streak с учетом релевантных дней ===
> function calculateStreak(entries, year, startDate, weekdays) {
>     // Фильтруем записи по году и дате начала
>     const filteredEntries = entries.filter(dateStr => {
>         const date = new Date(dateStr);
>         const isInYear = date.getFullYear() === year;
>         const isAfterStart = !startDate || date >= new Date(startDate);
>         return isInYear && isAfterStart;
>     });
>     
>     if (filteredEntries.length === 0) {
>         return { 
>             currentStreak: 0, 
>             recordStreak: 0, 
>             completedDays: 0 
>         };
>     }
>     
>     // Получаем все дни после даты начала
>     const allDays = getAllDaysAfterStart(year, startDate);
>     
>     // Получаем релевантные дни для проверки выполнения плана
>     const relevantDays = getRelevantDays(year, startDate, weekdays);
>     const relevantDaysSet = new Set(relevantDays.map(d => d.toISOString().split('T')[0]));
>     
>     // Создаем Set выполненных дней
>     const completedDates = new Set(filteredEntries.map(dateStr => {
>         const date = new Date(dateStr);
>         return date.toISOString().split('T')[0];
>     }));
>     
>     // Рассчитываем streak
>     let currentStreak = 0;
>     let recordStreak = 0;
>     let tempStreak = 0;
>     
>     // Сортируем все дни по дате
>     const sortedDays = [...allDays].sort((a, b) => a - b);
>     
>     for (let i = 0; i < sortedDays.length; i++) {
>         const day = sortedDays[i];
>         const dateStr = day.toISOString().split('T')[0];
>         const isCompleted = completedDates.has(dateStr);
>         const isRelevant = relevantDaysSet.has(dateStr);
>         
>         if (isCompleted) {
>             // Если день выполнен, увеличиваем временный streak
>             tempStreak++;
>             
>             // Проверяем, является ли этот день релевантным
>             // Если да, то это "обязательный" день, который должен быть выполнен
>             // Дополнительные дни (не релевантные) также увеличивают streak
>             
>             // Обновляем рекордный streak
>             if (tempStreak > recordStreak) {
>                 recordStreak = tempStreak;
>             }
>         } else {
>             // Если день не выполнен, проверяем, был ли он релевантным
>             if (isRelevant) {
>                 // Если пропущен релевантный день, сбрасываем streak
>                 tempStreak = 0;
>             }
>             // Если день не релевантный и не выполнен, не делаем ничего
>             // (streak не сбрасывается, но и не увеличивается)
>         }
>         
>         // Если это последний день, устанавливаем текущий streak
>         if (i === sortedDays.length - 1) {
>             currentStreak = tempStreak;
>         }
>     }
>     
>     return {
>         currentStreak,
>         recordStreak,
>         completedDays: filteredEntries.length
>     };
> }
> 
> // === Функция для расчета процента выполнения ===
> function calculateCompletionRate(entries, year, startDate, weekdays) {
>     const relevantDays = getRelevantDays(year, startDate, weekdays);
>     const totalRelevantDays = relevantDays.length;
>     
>     // Все выполненные дни после даты начала
>     const allCompletedDays = entries.filter(dateStr => {
>         const date = new Date(dateStr);
>         const isInYear = date.getFullYear() === year;
>         const isAfterStart = !startDate || date >= new Date(startDate);
>         return isInYear && isAfterStart;
>     }).length;
>     
>     // Записи, которые попадают в релевантные дни
>     const relevantEntries = entries.filter(dateStr => {
>         const date = new Date(dateStr);
>         const dateStrFormatted = date.toISOString().split('T')[0];
>         const isAfterStart = !startDate || date >= new Date(startDate);
>         return isAfterStart && relevantDays.some(day => 
>             day.toISOString().split('T')[0] === dateStrFormatted
>         );
>     });
>     
>     const completedRelevantDays = relevantEntries.length;
>     
>     return {
>         missedDays: totalRelevantDays - completedRelevantDays,
>         totalRelevantDays,
>         completionRate: totalRelevantDays > 0 ? Math.round((allCompletedDays / totalRelevantDays) * 100) : 0
>     };
> }
> 
> // === Обрабатываем каждую привычку ===
> const habitData = [];
> for (const page of habitPages) {
>     if (page.file.name === "Привычки") continue;
>     
>     const entries = page.entries || [];
>     const startDate = page.start_day;
>     const weekdays = page.weekday;
>     
>     // Streak с учетом релевантных дней
>     const streakInfo = calculateStreak(entries, TARGET_YEAR, startDate, weekdays);
>     
>     // Процент выполнения
>     const completionInfo = calculateCompletionRate(entries, TARGET_YEAR, startDate, weekdays);
>     
>     habitData.push({
>         name: page.file.name,
>         filePath: page.file.path,
>         currentStreak: streakInfo.currentStreak,
>         recordStreak: streakInfo.recordStreak,
>         completedDays: streakInfo.completedDays,
>         missedDays: completionInfo.missedDays,
>         completionRate: completionInfo.completionRate,
>         totalRelevantDays: completionInfo.totalRelevantDays,
>         hasFilters: !!startDate || (weekdays && weekdays.length > 0),
>         startDate: startDate,
>         weekdays: weekdays
>     });
> }
> 
> // === Создаем агрегированную привычку "Привычки" ===
> if (habitData.length > 0) {
>     const totalCompletedDays = habitData.reduce((sum, h) => sum + h.completedDays, 0);
>     const totalMissedDays = habitData.reduce((sum, h) => sum + h.missedDays, 0);
>     const totalCurrentStreak = Math.round(habitData.reduce((sum, h) => sum + h.currentStreak, 0) / habitData.length);
>     const totalRecordStreak = Math.round(habitData.reduce((sum, h) => sum + h.recordStreak, 0) / habitData.length);
>     
>     const totalRelevantDays = habitData.reduce((sum, h) => sum + h.totalRelevantDays, 0);
>     const totalCompletionRate = totalRelevantDays > 0 ? Math.round((totalCompletedDays / totalRelevantDays) * 100) : 0;
>     
>     habitData.push({
>         name: "Привычки",
>         currentStreak: totalCurrentStreak,
>         recordStreak: totalRecordStreak,
>         completedDays: totalCompletedDays,
>         missedDays: totalMissedDays,
>         completionRate: totalCompletionRate,
>         isAggregated: true
>     });
> }
> 
> // === Отображаем статистику ===
> if (habitData.length === 0) {
>     dv.paragraph("❌ Нет данных о привычках.");
> } else {
>     dv.el("h3", `Статистика привычек за ${TARGET_YEAR} год`);
>     
>     // Сортируем по текущему streak (по убыванию)
>     habitData.sort((a, b) => {
>         if (a.name === "Привычки") return 1;
>         if (b.name === "Привычки") return -1;
>         return b.currentStreak - a.currentStreak;
>     });
>     
>     // Детальная статистика
>     for (const habit of habitData) {
>         const isAggregated = habit.name === "Привычки";
>         const hasFilters = habit.hasFilters && !isAggregated;
>         const borderColor = isAggregated ? '#00ff00' : (hasFilters ? '#ff6d00' : '#8a2be2');
>         const titleColor = isAggregated ? '#00ff00' : (hasFilters ? '#ff6d00' : '#4fc3f7');
>         
>         let filterInfo = "";

>             const filterParts = [];

>                 filterParts.push(`<strong>Дата начала:</strong> ${formatDate(habit.startDate)}`);

>                 filterParts.push(`<strong>Дни недели:</strong> ${habit.weekdays.join(', ')}`);
>             }
>             
>             filterInfo = `<div style="
>                 color: #ffa726; 
>                 font-size: 11px; 
>                 margin-top: 5px;
>                 line-height: 1.4;
>                 background: rgba(255, 167, 38, 0.1);
>                 padding: 6px 8px;
>                 border-radius: 4px;
>             ">
>                 ⚙️ ${filterParts.join(' | ')}
>             </div>`;
>         }
>         
>         // Создаем кликабельное название привычки
>         let habitTitle;

>             habitTitle = `📊 ${habit.name} (итого)`;
