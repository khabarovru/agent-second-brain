```dataviewjs
const состояние = (dv.current().настроение + dv.current().эмоции + dv.current().самочувствие) / 3 * 10;
const дисциплина = (dv.current().подъем + dv.current().тренировка + dv.current().учеба + dv.current().таймменеджмент + dv.current().привычки) / 5 * 10;
const ответственность = (dv.current().финансы + dv.current().семья + dv.current().обещания + dv.current().обещания_себе) / 4 * 10;
const продуктивность = (dv.current().выполненные_задачи * dv.current().сложность) / (dv.current().всего_задач * 10) * 100;

const итог = (состояние + дисциплина + ответственность + продуктивность) / 4;

dv.paragraph(`📊 Итог дня: ${итог.toFixed(1)} / 100`);
```
