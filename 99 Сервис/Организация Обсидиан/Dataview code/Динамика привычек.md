// === Настройки ===
const FOLDER = "10 Организация Обсидиан/Привычки";
const MAX_STREAK = 7;

// === Расширенная палитра цветов ===
const colors = [
    'rgb(79, 195, 247)',   // голубой
    'rgb(138, 43, 226)',   // фиолетовый
    'rgb(255, 109, 0)',    // оранжевый
    'rgb(0, 255, 0)',      // зеленый
    'rgb(255, 255, 0)',    // желтый
    'rgb(255, 0, 255)',    // розовый
    'rgb(0, 255, 255)',    // бирюзовый
    'rgb(255, 69, 0)',     // красно-оранжевый
    'rgb(148, 0, 211)',    // темно-фиолетовый
    'rgb(50, 205, 50)',    // салатовый
    'rgb(30, 144, 255)',   // синий доджер
    'rgb(255, 20, 147)',   // глубокий розовый
    'rgb(64, 224, 208)',   // бирюзовый
    'rgb(255, 165, 0)',    // оранжевый
    'rgb(186, 85, 211)',   // орхидея
    'rgb(106, 90, 205)',   // сланец синий
    'rgb(72, 209, 204)',   // морская волна
    'rgb(199, 21, 133)',   // средний фиолетово-красный
    'rgb(70, 130, 180)',   // стальной синий
    'rgb(123, 104, 238)'   // средний сланец синий
];

// === Получаем все привычки из папки ===
const habitPages = dv.pages(`"${FOLDER}"`);

// === Текущий месяц ===
const now = new Date();
const currentYear = now.getFullYear();
const currentMonth = now.getMonth();
const startOfMonth = new Date(currentYear, currentMonth, 1);
const endOfMonth = new Date(currentYear, currentMonth + 1, 0);
const totalDaysInMonth = endOfMonth.getDate();

// === Функция для расчета streak за месяц ===
function calculateMonthlyStreak(entries, year, month) {
    const monthEntries = entries.filter(dateStr => {
        const date = new Date(dateStr);
        return date.getFullYear() === year && date.getMonth() === month;
    });
    
    const entryDates = new Set(monthEntries.map(dateStr => {
        const date = new Date(dateStr);
        return date.toISOString().split('T')[0];
    }));
    
    let currentStreak = 0;
    const currentDate = new Date(startOfMonth);
    const streakData = [];
    
    for (let i = 0; i < totalDaysInMonth; i++) {
        const dateStr = currentDate.toISOString().split('T')[0];
        
        if (entryDates.has(dateStr)) {
            currentStreak = Math.min(currentStreak + 1, MAX_STREAK);
        } else {
            currentStreak = Math.max(currentStreak - 1, 0);
        }
        
        streakData.push({
            date: dateStr,
            value: currentStreak,
            completed: entryDates.has(dateStr)
        });
        
        currentDate.setDate(currentDate.getDate() + 1);
    }
    
    return {
        currentStreak: streakData[streakData.length - 1].value,
        completedDays: monthEntries.length,
        missedDays: totalDaysInMonth - monthEntries.length,
        streakData: streakData
    };
}

// === Обрабатываем каждую привычку ===
const habitData = [];
for (const page of habitPages) {
    const entries = page.entries || [];
    const streakInfo = calculateMonthlyStreak(entries, currentYear, currentMonth);
    
    habitData.push({
        name: page.file.name,
        currentStreak: streakInfo.currentStreak,
        completedDays: streakInfo.completedDays,
        missedDays: streakInfo.missedDays,
        streakData: streakInfo.streakData,
        completionRate: Math.round((streakInfo.completedDays / totalDaysInMonth) * 100)
    });
}

// === Создаем график для текущего месяца ===
if (habitData.length === 0) {
    dv.paragraph("❌ Нет данных о привычках.");
} else {
    // Сортируем по текущему streak (по убыванию)
    habitData.sort((a, b) => b.currentStreak - a.currentStreak);
    
    // === График streak за месяц ===
    const chartData = {
        type: "line",
        data: {
            labels: Array.from({length: totalDaysInMonth}, (_, i) => i + 1),
            datasets: habitData.map((habit, index) => {
                // Используем индекс по модулю длины массива цветов для цикличного выбора
                const colorIndex = index % colors.length;
                
                return {
                    label: habit.name,
                    data: habit.streakData.map(d => d.value),
                    borderColor: colors[colorIndex],
                    backgroundColor: colors[colorIndex] + '20',
                    tension: 0.3,
                    pointRadius: 2,
                    pointHoverRadius: 4,
                    borderWidth: 2
                };
            })
        },
        options: {
            scales: {
                y: {
                    min: 0,
                    max: MAX_STREAK,
                    title: {
                        display: true,
                        text: "Streak (баллы)",
                        color: '#4fc3f7',
                        font: {
                            family: 'Orbitron, monospace',
                            size: 14
                        }
                    },
                    ticks: {
                        stepSize: 1,
                        color: '#e1e8ff'
                    },
                    grid: {
                        color: '#2a2a3a'
                    }
                },
                x: {
                    title: {
                        display: true,
                        text: "Дни месяца",
                        color: '#4fc3f7',
                        font: {
                            family: 'Orbitron, monospace',
                            size: 14
                        }
                    },
                    ticks: {
                        color: '#e1e8ff'
                    },
                    grid: {
                        color: '#2a2a3a'
                    }
                }
            },
            plugins: {
                title: {
                    display: true,
                    text: `Динамика привычек за ${now.toLocaleString('ru-RU', { month: 'long', year: 'numeric' })}`,
                    color: '#8a2be2',
                    font: {
                        family: 'Orbitron, monospace',
                        size: 18,
                        weight: 'bold'
                    }
                },
                legend: {
                    labels: {
                        color: '#e1e8ff',
                        font: {
                            family: 'Rajdhani, sans-serif',
                            size: 12
                        }
                    }
                },
                tooltip: {
                    backgroundColor: '#0d0d16',
                    titleColor: '#4fc3f7',
                    bodyColor: '#e1e8ff',
                    borderColor: '#8a2be2',
                    borderWidth: 1,
                    callbacks: {
                        label: function(context) {
                            const habit = habitData[context.datasetIndex];
                            const dayData = habit.streakData[context.dataIndex];
                            const date = new Date(dayData.date);
                            const dayOfMonth = date.getDate();
                            return `${context.dataset.label}: ${dayData.value} баллов (${dayData.completed ? '✅ выполнено' : '❌ пропущено'}) - ${dayOfMonth} число`;
                        }
                    }
                }
            },
            responsive: true,
            maintainAspectRatio: false
        }
    };
    
    // === Единый контейнер для графика ===
    const container = dv.el('div', '', {cls: 'habit-chart-container'});
    
    // Применяем стили через CSS класс
    container.style.cssText = `
        background: #0d0d16;
        border-radius: 8px;
        border: 1px solid #8a2be2;
        padding: 20px;
        margin: 16px 0;
        height: 400px;
        width: 100%;
        overflow: hidden;
    `;
    
    // Внутренний контейнер для графика
    const innerContainer = dv.el('div', '', {cls: 'habit-chart-inner'});
    innerContainer.style.cssText = `
        height: 350px;
        width: 100%;
        position: relative;
    `;
    
    container.appendChild(innerContainer);
    
    if (window.renderChart) {
        window.renderChart(chartData, innerContainer);
    } else {
        dv.paragraph("⚠️ Для отображения графика включи плагин Charts и опцию 'Enable DataviewJS'.");
    }
}
